function opt(v,cur){return`<option ${cur===v?'selected':''}>${v}</option>`;}
function fmtDate(ts){const d=new Date(ts);return d.toLocaleDateString('es-AR',{day:'2-digit',month:'2-digit',year:'2-digit'});}
function fmtDatetime(ts){const d=new Date(ts);return d.toLocaleDateString('es-AR',{day:'2-digit',month:'2-digit',year:'2-digit'})+' '+d.toLocaleTimeString('es-AR',{hour:'2-digit',minute:'2-digit'});}
function batColor(b){return b>=80?'#2E7D32':b>=50?'#E65100':'#C62828';}
function batBadge(b){const c=batColor(b);return`<div class="bat-wrap"><div class="bat-bar"><div class="bat-fill" style="width:${b}%;background:${c}"></div></div><span style="font-size:11px;color:${c};font-weight:600">${b}%</span></div>`;}
function stockBadge(s){return s===0?'<span class="sbadge s-out">Sin stock</span>':s<=2?'<span class="sbadge s-low">Bajo</span>':'<span class="sbadge s-ok">OK</span>';}
function totalAc(a){return(a.stockT||0)+(a.stockC||0);}
function stBadge(e){const m={Pendiente:'st-pend','En progreso':'st-prog','Listo para entregar':'st-list',Entregado:'st-ent'};return`<span class="st-badge ${m[e]||'st-pend'}">${e}</span>`;}
function canEdit(){return rol==='admin'||rol==='gerente';}
function rolLabel(){return rol==='admin'?'Admin':rol==='gerente'?'Gerente':'Empleado';}
function shortImei(i){return i?'···'+i.slice(-4):'-';}
function hl(text,q){if(!q)return text;const re=new RegExp(`(${q.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')})`,'gi');return String(text).replace(re,'<mark class="hl">$1</mark>');}
function matchEq(e,q){if(!q)return true;return[e.modelo,e.memoria,e.imei,e.sucursal,String(e.precio),String(e.bateria)].some(v=>v&&v.toLowerCase().includes(q));}
function matchAc(a,q){if(!q)return true;return[a.nombre,a.cat,String(a.precio)].some(v=>v&&v.toLowerCase().includes(q));}

function onSearch(tipo,inp){
  const q=inp.value.trim().toLowerCase();
  if(tipo==='eq'){searchEq=q;document.getElementById('clear-eq').style.display=q?'block':'none';renderEq();}
  else{searchAc=q;document.getElementById('clear-ac').style.display=q?'block':'none';renderAc();}
}
function clearSearch(tipo){
  if(tipo==='eq'){searchEq='';document.getElementById('search-eq').value='';document.getElementById('clear-eq').style.display='none';renderEq();}
  else{searchAc='';document.getElementById('search-ac').value='';document.getElementById('clear-ac').style.display='none';renderAc();}
}

function buildPagoSection(){
  return`<div class="pago-section">

      ${METODOS.map(m=>`<div class="pago-item">

function togglePago(id){const chk=document.getElementById('chk-'+id);const inp=document.getElementById('amt-'+id);inp.disabled=!chk.checked;if(!chk.checked)inp.value='';updateResumen();}
function updateResumen(){
  const t=document.getElementById('vtotal');if(!t)return;
  const ts=t.textContent||'0';const tn=parseFloat(ts.replace(/[^0-9.]/g,''))||0;
  const mon=ts.includes('USD')?'USD':'ARS';
  let pag=0;const act=[];
  METODOS.forEach(m=>{const chk=document.getElementById('chk-'+m.id);const amt=document.getElementById('amt-'+m.id);if(chk&&chk.checked&&amt){const v=parseFloat(amt.value)||0;pag+=v;if(v>0)act.push({...m,monto:v});}});
  const rEl=document.getElementById('pago-resumen');const mEl=document.getElementById('pago-msg');
  if(!act.length){rEl.style.display='none';mEl.innerHTML='';return;}
  rEl.style.display='block';const sym=mon==='USD'?'USD':'$';
  rEl.innerHTML=`<div class="pago-resumen">${act.map(a=>`<div class="pago-resumen-row"><span>${a.label}</span><strong>${a.prefix} ${Number(a.monto).toLocaleString('es-AR')}</strong></div>`).join('')}<div class="pago-resumen-total"><span>Total pagado</span><span>${sym} ${pag.toLocaleString('es-AR')}</span></div></div>`;
  const d=pag-tn;mEl.innerHTML=Math.abs(d)<0.01?`<div class="pago-ok">✓ Pago completo</div>`:d>0?`<div class="pago-ok">✓ Vuelto: ${sym} ${d.toLocaleString('es-AR')}</div>`:`<div class="pago-err">Falta cubrir: ${sym} ${Math.abs(d).toLocaleString('es-AR')}</div>`;
}
function getPagos(){return METODOS.filter(m=>{const c=document.getElementById('chk-'+m.id);const a=document.getElementById('amt-'+m.id);return c&&c.checked&&a&&parseFloat(a.value)>0;}).map(m=>({id:m.id,label:m.label.replace(/^.\s/,''),prefix:m.prefix,monto:parseFloat(document.getElementById('amt-'+m.id).value)||0}));}

/* ---- MODAL ---- */
function openModal(type,idx){
  modalType=type;editIdx=idx;
  const ttls={'eq-edit':idx===null?'Agregar equipo':'Editar equipo','ac-edit':idx===null?'Agregar accesorio':'Editar accesorio','venta':'Registrar venta','venta-eq':'Registrar venta — Equipo','venta-ac':'Registrar venta — Accesorio','venta-edit':'Editar venta','st-edit':idx===null?'Nuevo servicio técnico':'Editar servicio técnico'};
  document.getElementById('modal-ttl').textContent=ttls[type]||'';
  document.getElementById('modal-body').innerHTML=buildForm(type,idx);
  const mac=document.getElementById('modal-actions');
  if(type.startsWith('venta')&&type!=='venta-edit'){
    mac.innerHTML=`<button class="btn-cancel" onclick="closeModal()">Cancelar</button><button class="btn-save-green" onclick="saveModal()">Confirmar venta</button>`;
  } else {
    mac.innerHTML=`<button class="btn-cancel" onclick="closeModal()">Cancelar</button><button class="btn-save" onclick="saveModal()">Guardar</button>`;
  }
  if(type==='venta')setTimeout(updateVentaGeneral,50);
  document.getElementById('modal').style.display='flex';
}

        ${a.stockT>0?`<option value="Tunuyán">Tunuyán (${a.stockT||0} uds)</option>`:''}
        ${a.stockC>0?`<option value="La Consulta">La Consulta (${a.stockC||0} uds)</option>`:''}
      </select></div>
      <div class="mgrid">
        <div class="mfl"><label>CANTIDAD</label><input id="mf-cant" type="number" min="1" value="1" oninput="recalcTotal('ac',${idx})"/></div>
        <div class="mfl"><label>PRECIO VENTA (ARS $)</label><input id="mf-pventa" type="number" min="0" value="${a.precio}" oninput="recalcTotal('ac',${idx})"/></div>
      </div>
      <div class="sell-info" style="margin-bottom:12px">Total: <strong id="vtotal">$ ${a.precio.toLocaleString('es-AR')}</strong></div>
      <hr class="divider"/>${buildPagoSection()}`;
  }
  if(type==='venta'){
    const eqO=equipos.filter(e=>e.stock>0).map(e=>{const ri=equipos.indexOf(e);return`<option value="eq-${ri}">${e.modelo} ${e.memoria} · ${e.sucursal} (${e.stock}) — USD ${e.precio.toLocaleString('es-AR')}</option>`;}).join('');
    const acO=accesorios.filter(a=>totalAc(a)>0).map(a=>{const ri=accesorios.indexOf(a);return`<option value="ac-${ri}">${a.nombre} (${totalAc(a)}) — $ ${a.precio.toLocaleString('es-AR')}</option>`;}).join('');
    return`
      <div class="mfl"><label>PRODUCTO</label><select id="mf-prod" onchange="updateVentaGeneral()"><optgroup label="📱 Equipos">${eqO||'<option disabled>Sin stock</option>'}</optgroup><optgroup label="🎧 Accesorios">${acO||'<option disabled>Sin stock</option>'}</optgroup></select></div>
      <div class="mfl"><label>SUCURSAL DE VENTA</label><select id="mf-suc-venta">${SUCS.map(s=>opt(s,sucursal)).join('')}</select></div>
      <div class="mgrid">
        <div class="mfl"><label>CANTIDAD</label><input id="mf-cant" type="number" min="1" value="1" oninput="updateVentaGeneral()"/></div>
        <div class="mfl"><label>PRECIO VENTA</label><input id="mf-pventa" type="number" min="0" value="" oninput="updateVentaGeneral()"/></div>
      </div>
      <div class="sell-info" style="margin-bottom:12px">Total: <strong id="vtotal">—</strong></div>
      <hr class="divider"/>${buildPagoSection()}`;
  }
  if(type==='venta-edit'){
    const v=ventas[idx];

      <div class="sell-info"><strong>${v.producto}</strong><br>Sucursal: ${v.sucursal} · Cantidad: ${v.cantidad}</div>
      <div class="mfl"><label>SUCURSAL DE VENTA</label><select id="mf-suc-venta">${SUCS.map(s=>opt(s,v.sucursal)).join('')}</select></div>
      <div class="mgrid">
        <div class="mfl"><label>CANTIDAD</label><input id="mf-cant" type="number" min="1" value="${v.cantidad}" oninput="recalcEdit(${idx})"/></div>
        <div class="mfl"><label>PRECIO UNIT.</label><input id="mf-pventa" type="number" min="0" value="${v.precioUnit}" oninput="recalcEdit(${idx})"/></div>

    const hoy=new Date().toISOString().split('T')[0];

function recalcTotal(tipo,idx){
  const c=parseInt(document.getElementById('mf-cant')?.value)||0;
  const p=parseFloat(document.getElementById('mf-pventa')?.value)||0;
  const t=document.getElementById('vtotal');
  if(t){t.textContent=(tipo==='eq'?'USD ':'$ ')+(c*p).toLocaleString('es-AR');updateResumen();}
}
function recalcEdit(idx){
  const v=ventas[idx];const c=parseInt(document.getElementById('mf-cant')?.value)||0;const p=parseFloat(document.getElementById('mf-pventa')?.value)||0;
  const t=document.getElementById('vtotal');if(t)t.textContent=(v.moneda==='USD'?'USD ':'$ ')+(c*p).toLocaleString('es-AR');
}
function updateVentaGeneral(){
  const sel=document.getElementById('mf-prod')?.value;const c=parseInt(document.getElementById('mf-cant')?.value)||1;
  const pvEl=document.getElementById('mf-pventa');const t=document.getElementById('vtotal');
  if(!sel||!t)return;
  const [tipo,i]=sel.split('-');const idx=parseInt(i);
  const item=tipo==='eq'?equipos[idx]:accesorios[idx];
  if(pvEl&&!pvEl.value)pvEl.value=item.precio;
  const p=parseFloat(pvEl?.value)||item.precio;
  t.textContent=(tipo==='eq'?'USD ':'$ ')+(c*p).toLocaleString('es-AR');updateResumen();
}

function saveModal(){
  if(modalType==='eq-edit'){
    const modelo=document.getElementById('mf-modelo').value.trim();
    if(!modelo){alert('Ingresá el modelo');return;}
    const bat=parseInt(document.getElementById('mf-bat').value)||0;
    if(bat<1||bat>100){alert('Batería entre 1 y 100');return;}
    const imei=document.getElementById('mf-imei').value.trim();
    if(!imei){alert('Ingresá el IMEI');return;}
    const obj={modelo,memoria:document.getElementById('mf-mem').value,bateria:bat,imei,sucursal:document.getElementById('mf-suc').value,precio:parseInt(document.getElementById('mf-precio').value)||0,stock:parseInt(document.getElementById('mf-stock').value)||0};
    if(editIdx!==null)equipos[editIdx]=obj;else equipos.push(obj);
    renderEq();closeModal();return;
  }
  if(modalType==='ac-edit'){
    const nombre=document.getElementById('mf-nombre').value.trim();
    if(!nombre){alert('Ingresá el nombre');return;}
    const obj={nombre,cat:document.getElementById('mf-cat-ac').value,precio:parseInt(document.getElementById('mf-precio').value)||0,stockT:parseInt(document.getElementById('mf-stockT').value)||0,stockC:parseInt(document.getElementById('mf-stockC').value)||0};
    if(editIdx!==null)accesorios[editIdx]=obj;else accesorios.push(obj);
    renderAc();closeModal();return;
  }
  const regVenta=(tipo,prodIdx,c,p,sv,moneda,producto,sStock)=>{
    ventas.push({ts:Date.now(),tipo,producto,sucursal:sv,stockSucursal:sStock,cantidad:c,precioUnit:p,total:c*p,moneda,por:rolLabel(),estado:'activa',eqIdx:tipo==='equipo'?prodIdx:null,acIdx:tipo==='accesorio'?prodIdx:null,pagos:getPagos()});
  };
  if(modalType==='venta-eq'){
    const e=equipos[editIdx];const c=parseInt(document.getElementById('mf-cant').value)||1;const p=parseFloat(document.getElementById('mf-pventa').value)||e.precio;const sv=document.getElementById('mf-suc-venta').value;
    if(c>e.stock){alert(`Stock insuficiente. Disponible: ${e.stock}`);return;}
    equipos[editIdx].stock-=c;regVenta('equipo',editIdx,c,p,sv,'USD',`${e.modelo} ${e.memoria}`,e.sucursal);
    renderEq();renderVentas();closeModal();return;
  }
  if(modalType==='venta-ac'){
    const a=accesorios[editIdx];const c=parseInt(document.getElementById('mf-cant').value)||1;const p=parseFloat(document.getElementById('mf-pventa').value)||a.precio;
    const sv=document.getElementById('mf-suc-venta').value;
    const maxSt=sv==='Tunuyán'?(a.stockT||0):(a.stockC||0);
    if(c>maxSt){alert(`Stock insuficiente en ${sv}. Disponible: ${maxSt}`);return;}
    if(sv==='Tunuyán')accesorios[editIdx].stockT-=c;else accesorios[editIdx].stockC-=c;
    regVenta('accesorio',editIdx,c,p,sv,'ARS',a.nombre,sv);
    renderAc();renderVentas();closeModal();return;
  }
  if(modalType==='venta'){
    const sel=document.getElementById('mf-prod').value;const c=parseInt(document.getElementById('mf-cant').value)||1;const p=parseFloat(document.getElementById('mf-pventa').value)||0;const sv=document.getElementById('mf-suc-venta').value;
    if(!sel){alert('Seleccioná un producto');return;}
    const [tipo,i]=sel.split('-');const idx=parseInt(i);
    if(tipo==='eq'){
      const e=equipos[idx];if(c>e.stock){alert(`Stock insuficiente. Disponible: ${e.stock}`);return;}
      equipos[idx].stock-=c;regVenta('equipo',idx,c,p||e.precio,sv,'USD',`${e.modelo} ${e.memoria}`,e.sucursal);renderEq();
    } else {
      const a=accesorios[idx];if(c>totalAc(a)){alert(`Stock insuficiente. Disponible: ${totalAc(a)}`);return;}
      // descontar del mayor stock disponible
      if((a.stockT||0)>=(a.stockC||0))a.stockT=Math.max(0,(a.stockT||0)-c);else a.stockC=Math.max(0,(a.stockC||0)-c);
      regVenta('accesorio',idx,c,p||a.precio,sv,'ARS',a.nombre,sv);renderAc();
    }
    renderVentas();closeModal();return;
  }
  if(modalType==='venta-edit'){
    const v=ventas[editIdx];const cN=parseInt(document.getElementById('mf-cant').value)||1;const pN=parseFloat(document.getElementById('mf-pventa').value)||v.precioUnit;const svN=document.getElementById('mf-suc-venta').value;
    if(v.tipo==='equipo'&&v.eqIdx!=null){equipos[v.eqIdx].stock+=v.cantidad;if(cN>equipos[v.eqIdx].stock){alert(`Stock insuficiente`);equipos[v.eqIdx].stock-=v.cantidad;return;}equipos[v.eqIdx].stock-=cN;}
    ventas[editIdx]={...v,cantidad:cN,precioUnit:pN,total:cN*pN,sucursal:svN};
    renderEq();renderAc();renderVentas();closeModal();return;
  }
  if(modalType==='st-edit'){
    const nombre=document.getElementById('st-nombre').value.trim();
    if(!nombre){alert('Ingresá el nombre del cliente');return;}
    const modelo=document.getElementById('st-modelo').value.trim();
    if(!modelo){alert('Ingresá el modelo del equipo');return;}
    const fechaVal=document.getElementById('st-fecha').value;
    const obj={
      nombre,
      telefono:document.getElementById('st-tel').value.trim(),
      correo:document.getElementById('st-correo').value.trim(),
      modelo,
      icloud:document.getElementById('st-icloud').value.trim(),
      codigo:document.getElementById('st-codigo').value.trim(),
      detalle:document.getElementById('st-detalle').value.trim(),
      precio:parseInt(document.getElementById('st-precio').value)||0,
      estado:document.getElementById('st-estado').value,
      sucursal:document.getElementById('st-suc').value,
      fechaIngreso:fechaVal?new Date(fechaVal+'T00:00:00').getTime():Date.now(),
    };
    if(editIdx!==null)servicios[editIdx]=obj;else servicios.push(obj);
    renderST();closeModal();return;
  }
}

function anularVenta(idx){
  if(!confirm('¿Anular esta venta? El stock será devuelto.'))return;
  const v=ventas[idx];
  if(v.tipo==='equipo'&&v.eqIdx!=null)equipos[v.eqIdx].stock+=v.cantidad;
  if(v.tipo==='accesorio'&&v.acIdx!=null){const a=accesorios[v.acIdx];if(v.sucursal==='Tunuyán')a.stockT+=v.cantidad;else a.stockC+=v.cantidad;}
  ventas[idx]={...v,estado:'anulada'};
  renderEq();renderAc();renderVentas();
}
function delItem(type,idx){
  if(!confirm('¿Eliminar?'))return;
  if(type==='eq'){equipos.splice(idx,1);renderEq();}
  else if(type==='ac'){accesorios.splice(idx,1);renderAc();}
  else{servicios.splice(idx,1);renderST();}
}
function closeModal(){document.getElementById('modal').style.display='none';}
